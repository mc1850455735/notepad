- 暴力

直接建一个大小为n - 1的hash表, 观察插入哪个时太大了

```cpp
class Solution {
public:
    vector<int> getSneakyNumbers(vector<int>& nums) {
        vector<int> res;
        unordered_set<int> t;
        for(int i = 0; i < nums.size(); i++) {
            if(t.count(nums[i])) {
                res.push_back(nums[i]);
            }
            t.insert(nums[i]);
        }

        return res;
    }
};
```

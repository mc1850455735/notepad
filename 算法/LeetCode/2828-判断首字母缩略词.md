```cpp
class Solution {
public:
    bool isAcronym(vector<string>& words, string s) {
        std::vector<char> tt;
        for(int i = 0; i < words.size(); i++) {
            tt.push_back(words[i][0]);
        }
        bool isRes = true;
        if(s.length() != tt.size())
            return false;
        for(int i = 0; i < tt.size(); i++) {
            if(s[i] != tt[i]){
                isRes = false;
                break;
            }
        }

        return isRes;
    }
};
```
